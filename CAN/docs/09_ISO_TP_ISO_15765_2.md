# ISO-TP (ISO 15765-2): Detailed Description

## Overview

ISO-TP (ISO 15765-2) is a **transport protocol** that operates on top of CAN to enable the transmission of messages **larger than 8 bytes**. Since standard CAN frames are limited to 8 bytes of payload, ISO-TP provides segmentation and reassembly mechanisms to handle larger data transfers, commonly used in automotive diagnostics (UDS - Unified Diagnostic Services).

## Key Concepts

### Protocol Data Unit (PDU) Types

ISO-TP defines four frame types for message transfer:

1. **Single Frame (SF)**: Contains the entire message (≤7 bytes for standard CAN)
2. **First Frame (FF)**: First segment of a multi-frame message
3. **Consecutive Frame (CF)**: Subsequent segments
4. **Flow Control (FC)**: Receiver feedback for flow management

### Frame Structure

```
Single Frame (SF):
[0]: 0x0N (N = data length)
[1-7]: Data bytes

First Frame (FF):
[0]: 0x1N (N = upper nibble of length)
[1]: Lower byte of length
[2-7]: First 6 data bytes

Consecutive Frame (CF):
[0]: 0x2N (N = sequence number 0-15)
[1-7]: Data bytes

Flow Control (FC):
[0]: 0x3X (X = 0:CTS, 1:Wait, 2:Overflow)
[1]: Block size (0 = unlimited)
[2]: Separation time (STmin)
```

#define ISOTP_MAX_DLEN 4095
#define ISOTP_BUFSIZE 4096

// ISO-TP frame types
typedef enum {
    ISOTP_SINGLE = 0,
    ISOTP_FIRST = 1,
    ISOTP_CONSECUTIVE = 2,
    ISOTP_FLOW_CONTROL = 3

    FC_CTS = 0,      // Continue to send
    FC_WAIT = 1,     // Wait
    FC_OVERFLOW = 2  // Buffer overflow
} isotp_fc_state_t;

    uint8_t rx_buffer[ISOTP_BUFSIZE];
    uint8_t tx_buffer[ISOTP_BUFSIZE];
    uint16_t rx_len;
    uint16_t tx_len;

    memset(ctx, 0, sizeof(isotp_context_t));

    return 0;
}

// Send first frame
int isotp_send_first_frame(isotp_context_t *ctx, uint32_t can_id, 
                           const uint8_t *data, uint16_t len) {
    if (len > ISOTP_MAX_DLEN) return -1;
    
    uint8_t frame[8];
    frame[0] = 0x10 | ((len >> 8) & 0x0F);  // FF with upper length nibble
    frame[1] = len & 0xFF;                   // Lower length byte
    memcpy(&frame[2], data, 6);              // First 6 bytes of data
    
    ctx->tx_idx = 6;
    ctx->tx_sn = 1;
    ctx->tx_active = true;
    
    // can_send(can_id, frame, 8);
    
    return 0;
}

// Send consecutive frame
int isotp_send_consecutive_frame(isotp_context_t *ctx, uint32_t can_id, 
                                 const uint8_t *data, uint16_t total_len) {
    uint8_t frame[8];
    frame[0] = 0x20 | (ctx->tx_sn & 0x0F);  // CF with sequence number
    
    uint16_t remaining = total_len - ctx->tx_idx;
    uint8_t copy_len = (remaining > 7) ? 7 : remaining;
    
    memcpy(&frame[1], &data[ctx->tx_idx], copy_len);
    
    ctx->tx_idx += copy_len;
    ctx->tx_sn = (ctx->tx_sn + 1) & 0x0F;
    
    if (ctx->tx_idx >= total_len) {
        ctx->tx_active = false;
    }
    
    // can_send(can_id, frame, 8);
    
    return 0;
}

// Send flow control frame
int isotp_send_flow_control(uint32_t can_id, isotp_fc_state_t state, 
                            uint8_t block_size, uint8_t stmin) {
    uint8_t frame[8] = {0};
    frame[0] = 0x30 | (state & 0x0F);
    frame[1] = block_size;
    frame[2] = stmin;
    
    // can_send(can_id, frame, 8);
    
    return 0;
}

// Process received CAN frame
int isotp_receive(isotp_context_t *ctx, const uint8_t *frame, uint8_t len) {
    uint8_t frame_type = (frame[0] >> 4) & 0x0F;
    
    switch (frame_type) {
        case ISOTP_SINGLE: {
            uint8_t data_len = frame[0] & 0x0F;
            if (data_len > 7) return -1;
            
            memcpy(ctx->rx_buffer, &frame[1], data_len);
            ctx->rx_len = data_len;
            ctx->rx_active = false;
            return data_len;  // Complete message received
        }
        
        case ISOTP_FIRST: {
            uint16_t data_len = ((frame[0] & 0x0F) << 8) | frame[1];
            if (data_len > ISOTP_MAX_DLEN) return -1;
            
            ctx->rx_len = data_len;
            ctx->rx_idx = 6;
            ctx->rx_sn = 1;
            ctx->rx_active = true;
            
            memcpy(ctx->rx_buffer, &frame[2], 6);
            
            // Send flow control - continue to send
            // isotp_send_flow_control(response_id, FC_CTS, 0, 0);
            return 0;  // More frames expected
        }
        
        case ISOTP_CONSECUTIVE: {
            if (!ctx->rx_active) return -1;
            
            uint8_t sn = frame[0] & 0x0F;
            if (sn != ctx->rx_sn) return -1;  // Sequence error
            
            uint16_t remaining = ctx->rx_len - ctx->rx_idx;
            uint8_t copy_len = (remaining > 7) ? 7 : remaining;
            
            memcpy(&ctx->rx_buffer[ctx->rx_idx], &frame[1], copy_len);
            ctx->rx_idx += copy_len;
            ctx->rx_sn = (ctx->rx_sn + 1) & 0x0F;
            
            if (ctx->rx_idx >= ctx->rx_len) {
                ctx->rx_active = false;
                return ctx->rx_len;  // Complete message received
            }
            return 0;  // More frames expected
        }
        
        case ISOTP_FLOW_CONTROL: {
            uint8_t fc_state = frame[0] & 0x0F;
            ctx->block_size = frame[1];
            ctx->stmin = frame[2];
            
            if (fc_state == FC_CTS && ctx->tx_active) {
                // Continue sending consecutive frames
                return 1;  // Ready to send next CF
            }
            return 0;
        }
        
        default:
            return -1;
    }
}

// Example usage
void isotp_example(void) {
    isotp_context_t ctx;
    isotp_init(&ctx);
    
    // Send a short message (single frame)
    uint8_t short_msg[] = {0x01, 0x02, 0x03, 0x04};
    isotp_send_single_frame(&ctx, 0x7E0, short_msg, 4);
    
    // Send a long message (multi-frame)
    uint8_t long_msg[100];
    for (int i = 0; i < 100; i++) {
        long_msg[i] = i;
    }
    
    memcpy(ctx.tx_buffer, long_msg, 100);
    ctx.tx_len = 100;
    isotp_send_first_frame(&ctx, 0x7E0, long_msg, 100);
    
    // After receiving flow control, send consecutive frames
    while (ctx.tx_active) {
        isotp_send_consecutive_frame(&ctx, 0x7E0, long_msg, 100);
        // Add timing delay based on stmin
    }
}

const ISOTP_MAX_DLEN: usize = 4095;

    // Send first frame and prepare for consecutive frames
    pub fn send_first_frame(&mut self, can_id: u32, data: &[u8]) -> Result<Vec<u8>, &'static str> {
        if data.len() > ISOTP_MAX_DLEN {
            return Err("Data exceeds maximum ISO-TP length");
        }

        let mut frame = vec![0u8; 8];
        let len = data.len() as u16;
        
        frame[0] = 0x10 | ((len >> 8) & 0x0F) as u8;

    pub fn send_consecutive_frame(&mut self, can_id: u32) -> Result<Option<Vec<u8>>, &'static str> {
        if !self.tx_active {
            return Ok(None);
        }

        let mut frame = vec![0u8; 8];
        frame[0] = 0x20 | (self.tx_sn & 0x0F);

        let remaining = self.tx_len - self.tx_idx;
        let copy_len = min(remaining as usize, 7);

        frame[1..=copy_len].copy_from_slice(
            &self.tx_buffer[self.tx_idx as usize..(self.tx_idx as usize + copy_len)]
        );

        self.tx_idx += copy_len as u16;
        self.tx_sn = (self.tx_sn + 1) & 0x0F;

        if self.tx_idx >= self.tx_len {
            self.tx_active = false;
        }

        Ok(Some(frame))
    }

    // Send flow control
    pub fn send_flow_control(
        state: FlowControlState,
        block_size: u8,
        stmin: u8,
    ) -> Vec<u8> {
        let mut frame = vec![0u8; 8];
        frame[0] = 0x30 | (state as u8);
        frame[1] = block_size;
        frame[2] = stmin;
        frame
    }

    // Receive and process CAN frame
    pub fn receive(&mut self, frame: &[u8]) -> Result<Option<Vec<u8>>, &'static str> {
        if frame.is_empty() {
            return Err("Empty frame");
        }

        let frame_type = (frame[0] >> 4) & 0x0F;

        match frame_type {
            0 => self.receive_single_frame(frame),
            1 => self.receive_first_frame(frame),
            2 => self.receive_consecutive_frame(frame),
            3 => self.receive_flow_control(frame),
            _ => Err("Invalid frame type"),
        }
    }

    fn receive_single_frame(&mut self, frame: &[u8]) -> Result<Option<Vec<u8>>, &'static str> {
        let data_len = (frame[0] & 0x0F) as usize;

        self.rx_buffer[..data_len].copy_from_slice(&frame[1..=data_len]);
        self.rx_len = data_len as u16;
        self.rx_active = false;

        Ok(Some(self.rx_buffer[..data_len].to_vec()))
    }

    fn receive_first_frame(&mut self, frame: &[u8]) -> Result<Option<Vec<u8>>, &'static str> {

            return Err("Message too long");

        self.rx_len = data_len;
        self.rx_idx = 6;
        self.rx_sn = 1;
        self.rx_active = true;

        self.rx_buffer[..6].copy_from_slice(&frame[2..8]);

    fn receive_consecutive_frame(&mut self, frame: &[u8]) -> Result<Option<Vec<u8>>, &'static str> {
        if !self.rx_active {
            return Err("Not expecting consecutive frame");
        }

        let sn = frame[0] & 0x0F;
        if sn != self.rx_sn {
            return Err("Sequence number mismatch");
        }

        let remaining = self.rx_len - self.rx_idx;
        let copy_len = min(remaining as usize, 7);

        self.rx_buffer[self.rx_idx as usize..(self.rx_idx as usize + copy_len)]
            .copy_from_slice(&frame[1..=copy_len]);

        self.rx_idx += copy_len as u16;
        self.rx_sn = (self.rx_sn + 1) & 0x0F;

        if self.rx_idx >= self.rx_len {
            self.rx_active = false;
            return Ok(Some(self.rx_buffer[..self.rx_len as usize].to_vec()));
        }

        Ok(None) // More frames expected
    }

    fn receive_flow_control(&mut self, frame: &[u8]) -> Result<Option<Vec<u8>>, &'static str> {
        let fc_state = frame[0] & 0x0F;
        self.block_size = frame[1];
        self.stmin = frame[2];

        if fc_state == 0 && self.tx_active {
            // Continue to send - ready for next consecutive frame
            return Ok(None);
        }

        Ok(None)
    }
}

## Summary

**ISO-TP (ISO 15765-2)** is a crucial transport layer protocol for CAN networks that enables transmission of messages exceeding the 8-byte CAN payload limit. It provides:

- **Segmentation/Reassembly**: Breaks large messages into multiple CAN frames and reconstructs them at the receiver
- **Four Frame Types**: Single Frame (≤7 bytes), First Frame (initiates multi-frame), Consecutive Frame (continuation), and Flow Control (receiver management)
- **Sequence Management**: Uses sequence numbers (0-15) to ensure correct frame ordering
- **Flow Control**: Receiver can pace transmission using block size and minimum separation time parameters
- **Maximum Length**: Supports messages up to 4095 bytes

Common applications include automotive diagnostics (UDS protocol), ECU programming, and any scenario requiring large data transfers over CAN. The protocol is widely used in OBDII diagnostics, where diagnostic requests and responses often exceed 8 bytes.
